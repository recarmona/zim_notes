**Borrar un namespace atascado**


kubectl get namespace "bigbang" -o json \
  | tr -d "\n" | sed "s/\"finalizers\": \[[^]]\+\]/\"finalizers\": []/" \
  | kubectl replace --raw /api/v1/namespaces/bigbang/finalize -f -
